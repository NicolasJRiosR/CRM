jwtInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { const auth = inject(AuthService); const token = auth.tokenSig(); if (!token) return next(req); const authReq = req.clone({ setHeaders: { Authorization: `Bearer ${token}` }, }); return next(authReq); }
